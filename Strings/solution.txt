#include <stdio.h>
#include<string.h>
int main() {
    int T;
    scanf("%d",&T);
    while(T--){
        char str[1000];
        scanf("%s",str);
        int n = strlen(str);
        for(int i = 0 ; i < n ;i++){
            if(str[i]>=97 && str[i] <=109){
                int t = 'z' - str[i];
                t++;
                t = 26 - t;
                str[i] = 'z' - t;
            }else if(str[i]>=110 && str[i] <=122){
                int t = 'z' - str[i];
                t = 'a' + t; 
                str[i] = t;
            }
        }
        printf("%s\n",str);
    }
	//code
	return 0;
}
